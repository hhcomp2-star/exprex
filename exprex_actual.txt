CREATE TABLE conductores (
            cedula TEXT PRIMARY KEY,
            vehiculo TEXT,
            placa TEXT, 
            propio TEXT DEFAULT 'No',
            disponible TEXT DEFAULT 'Sí',
            vence_certificado TEXT,
            vence_rotc TEXT,
            capacidad_carga TEXT,
            FOREIGN KEY (cedula) REFERENCES usuarios (cedula)
        );
CREATE TABLE finanzas_personal (
            cedula TEXT PRIMARY KEY,
            tipo_pago TEXT DEFAULT 'Sueldo Fijo',
            monto_base REAL DEFAULT 0.0,
            cestaticket REAL DEFAULT 0.0,
            pago_por_viaje REAL DEFAULT 0.0,
            bono_empresa REAL DEFAULT 0.0,
            bono_decreto REAL DEFAULT 0.0,
            otros_ingresos REAL DEFAULT 0.0,
            FOREIGN KEY (cedula) REFERENCES usuarios (cedula)
        );
CREATE TABLE control_combustible (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            placa TEXT,                      -- Rastrea el camión físico
            cedula TEXT,                     -- Rastrea al humano que reporta
            fecha TEXT,                      -- Formato AAAA-MM-DD
            km_actual INTEGER,               -- Odómetro de la unidad
            litros_comprados REAL,
            costo_usd REAL,
            tasa_cambio REAL,                -- Tasa oficial BCV
            costo_bs REAL,                   -- Multiplicación automática por código
            estacion_servicio TEXT,
            FOREIGN KEY (cedula) REFERENCES usuarios (cedula)
        );
CREATE TABLE sqlite_sequence(name,seq);
CREATE TABLE combustible (
            id_registro INTEGER PRIMARY KEY AUTOINCREMENT,
            cedula TEXT NOT NULL,
            fecha TEXT NOT NULL,
            km_actual REAL NOT NULL,
            litros_comprados REAL NOT NULL,
            costo_usd REAL NOT NULL,
            tasa_cambio REAL NOT NULL,
            costo_bs REAL NOT NULL,
            estacion_servicio TEXT,
            FOREIGN KEY (cedula) REFERENCES usuarios (cedula) ON DELETE RESTRICT
        );
CREATE TABLE clientes (id_cliente INTEGER PRIMARY KEY AUTOINCREMENT, rif TEXT UNIQUE NOT NULL, razon_social TEXT NOT NULL, telefono_contacto TEXT, email_contacto TEXT, direccion_fiscal TEXT);
CREATE TABLE vehiculos (
    placa TEXT PRIMARY KEY,
    marca TEXT NOT NULL,
    modelo TEXT NOT NULL,
    anio INTEGER,
    kilometraje_actual REAL DEFAULT 0,
    serial_motor TEXT,
    serial_carroceria TEXT,
    capacidad_peso_kg REAL DEFAULT 0,
    capacidad_volumen_m3 REAL DEFAULT 0,
    vencimiento_rcv TEXT, -- ✨ ¡Corregido a RCV!
    vencimiento_trimestres TEXT,
    chofer_asignado TEXT,
    propietario_tipo TEXT DEFAULT 'Propio',
    propietario_cedula TEXT,
    propietario_nombre TEXT,
    estatus TEXT DEFAULT 'Operativo',
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE gastos (
    id_gasto INTEGER PRIMARY KEY AUTOINCREMENT,
    fecha TEXT NOT NULL,
    categoria TEXT NOT NULL,
    monto_usd REAL NOT NULL,
    observaciones TEXT DEFAULT ''
, placa_vehiculo TEXT DEFAULT 'General');
CREATE TABLE configuracion (
    clave TEXT PRIMARY KEY,
    valor TEXT NOT NULL
);
CREATE TABLE gastos_operativos_viaje (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    fecha TEXT NOT NULL,                   -- Fecha del gasto
    placa TEXT NOT NULL,                   -- Qué unidad generó el gasto
    tipo_gasto TEXT NOT NULL,              -- 'Viáticos', 'Peajes', 'Imprevisto Mecánico', 'Aseguramiento de Carga'
    monto_bs REAL NOT NULL,                -- Monto en Bolívares de la factura/recibo
    tasa_cambio REAL NOT NULL,             -- Tasa BCV automática
    monto_usd REAL NOT NULL,               -- Equivalente en $
    estacion_origen_destino TEXT,          -- Ej: "Caracas - Valencia"
    observaciones TEXT                     -- Detalles (Ej: "Almuerzo de la ruta y 4 peajes")
);
CREATE TABLE cxc_independiente (
    id_cxc INTEGER PRIMARY KEY AUTOINCREMENT,
    deudor TEXT,
    concepto TEXT,
    monto_inicial REAL,
    monto_pendiente REAL,
    fecha_registro TEXT,
    estatus TEXT DEFAULT 'Pendiente',
    notas TEXT
);
CREATE TABLE cxp_independiente (
    id_cxp INTEGER PRIMARY KEY AUTOINCREMENT,
    acreedor TEXT,
    concepto TEXT,
    monto_inicial REAL,
    monto_pendiente REAL,
    fecha_registro TEXT,
    estatus TEXT DEFAULT 'Pendiente',
    notas TEXT
);
CREATE TABLE sucursales (
    id_sucursal INTEGER PRIMARY KEY AUTOINCREMENT,
    id_cliente INTEGER NOT NULL,
    nombre_agencia TEXT NOT NULL,
    ciudad TEXT NOT NULL,
    direccion TEXT NOT NULL,
    telefono_sucursal TEXT,
    latitud REAL NOT NULL DEFAULT 10.4985,
    longitud REAL NOT NULL DEFAULT -66.9015,
    activa TEXT NOT NULL DEFAULT 'Sí'
);
CREATE TABLE viajes (
    id_viaje INTEGER PRIMARY KEY AUTOINCREMENT,
    id_cliente INTEGER,
    id_sucursal_origen INTEGER,
    cedula_conductor TEXT,
    fecha_despacho TEXT,
    origen TEXT,
    destino TEXT,
    tipo_material TEXT,
    distancia_km REAL,
    tipo_viaje TEXT,
    peso_carga_kg REAL,
    monto_flete_usd REAL,
    persona_contacto_entrega TEXT,
    telefono_contacto_entrega TEXT,
    observaciones TEXT,
    num_pedido TEXT,
    num_factura TEXT,
    estatus_viaje TEXT,
    cliente_solicitante TEXT,
    telefono_cliente TEXT,
    latitud_entrega REAL,
    longitud_entrega REAL,
    foto_evidencia TEXT,
    descuento_usd REAL,
    importe_neto_usd REAL,
    pago_chofer_usd REAL,
    beneficio_exprex_usd REAL
);
CREATE TABLE IF NOT EXISTS "usuarios" (
"cedula" TEXT,
  "nombre" TEXT,
  "telefono" TEXT,
  "email" TEXT,
  "direccion" TEXT,
  "contrasena" TEXT,
  "rol" TEXT,
  "activo" TEXT,
  "fecha_baja" TEXT,
  "departamento" TEXT,
  "banco" TEXT,
  "numero_cuenta" TEXT
);
